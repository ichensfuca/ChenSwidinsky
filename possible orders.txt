
// possible orders for the deviation group
list_divisors:={};
for n in [1..20] do
  bb:=(6*16^(n-1))^2;
  new_divisors:=[d : d in Divisors(bb) | d le 255];
  list_divisors:=list_divisors join Set(new_divisors);
end for;
list_divisors;

assert list_divisors eq { 1, 2, 3, 4, 6, 8, 9, 12, 16, 18, 24, 32, 36, 48, 64, 72, 96, 128, 144, 192 };
